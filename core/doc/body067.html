BaoとはBusinessLogic Access Objectの略です
S2Daoの様にinterfaceを定義してAOPを設定するだけでs2-buriの持つすべての機能が簡単に利用できます。
仮にXPDLが未実装でもS2のモック機能を使ってモックとして動作させることも出来ます。

// Bao用interfaceの終了はBaoで終了させます。
public interface OrderBao {

// このinterfaceが対象にするプロセス名
public static String PROCESS = "注文管理.注文";

// データ取得用メソッドには必ずACTIVITYを付けてください
public static String getUnderWork_ACTIVITY = "出荷作業中";
// メソッド名はget/select/find/countで始めるとデータ取得用と判断します。
// デフォルトで引数が一つの場合は#dateとしてアクセスします
public List getUnderWork(HogeFindDto dto);

// これはプロセスの実行用メソッドになります。Activity指定は省略可能です。(自動的に最適なActivityを選択します)
public void order(HogeDto dto);

//ユーザ情報を設定したい場合、StandardInvokerを自動的に使う
//引数の内容としてはdata,userInfoしか認識しない。省略時はdata
//他の名前を付けたモノは自動的にContextで使えるようになる
public static String order_ARGS = "data,userInfo";
public void order(HogeDto dto,HogeID id);

//#actionの設定方法
public static String cancel_ACTION = "cancel";
//戻り値の設定方法
public static String cancel_RESULT = "#result";
public Stirng cancel(HogeDto dto);
